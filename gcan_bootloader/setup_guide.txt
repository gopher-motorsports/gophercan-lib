



# Ensure the following programs are installed
# If STM_Programmer_CLI is not found, try adding it to your PATH variables (Should happen automatically)

STM32CubeProgrammer 2.19
Jinja2==3.1.2
PyYAML==6.0

# To auto generate hex and bin build outputs, add the following to your CMakeLists.txt:

# Convert output to hex and binary
add_custom_command(TARGET ${CMAKE_PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:${CMAKE_PROJECT_NAME}> ${CMAKE_PROJECT_NAME}.hex
)

# Convert to bin file
add_custom_command(TARGET ${CMAKE_PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${CMAKE_PROJECT_NAME}> ${CMAKE_PROJECT_NAME}.bin
)
